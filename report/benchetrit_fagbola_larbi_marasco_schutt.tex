\begin{itemize}
  \item \textbf{Report scope}
  \begin{itemize}
    \item dataset: 1y inflation caps, floors, swaps; areas: EU, US
    \item objective: option price time series $\rightarrow$ implied moments (mean, variance, skewness, kurtosis)
    \item focus: identification, stability; no forecasting
  \end{itemize}

  \item \textbf{Deliverables}
  \begin{itemize}
    \item code: \texttt{code in github}
    \item outputs: \texttt{moments\_main.csv}, \texttt{moments\_direct\_bkm.csv}, \texttt{moments\_parametric\_lognormal.csv}, \texttt{moments\_parametric\_normal.csv}
  \end{itemize}

  \item \textbf{LLM disclosure}

    \item ChatGPT (GPT-5.2 Thinking)
    \item usage: diagnostic narratives and LaTeX bullet structuring
    \item no usage: data generation, result fabrication, code execution on hidden data

  \item \textbf{Literature review: method families \& mappings}
  \begin{enumerate}
    \item \textbf{Prices $\rightarrow$ pdf (risk-neutral density / state prices)}
    \begin{itemize}
      \item Breeden--Litzenberger (1978) (nonparametric; differentiation of call price curve; $\sim$3315 cites)
      \item Jackwerth--Rubinstein (1996) (nonparametric; constrained recovery / smoothing; $\sim$1614 cites)
      \item A\"it-Sahalia--Lo (1998/1995) (nonparametric; kernel / smoothing; $\sim$1591 cites)
      \item spline/shape-constrained smoothing (convexity/monotonicity enforcement)
      \item parametric densities (lognormal / mixtures) (pdf implied by calibrated parameters)
    \end{itemize}

    \item \textbf{Pdf $\rightarrow$ moments (mean/var/skew/kurt)}
    \begin{itemize}
      \item numerical quadrature on grid (trapz) on truncated support
      \item analytic moments (parametric families; e.g., lognormal moments)
      \item parametric fit to recovered pdf (diagnostic; e.g., Normal fit on grid)
    \end{itemize}

    \item \textbf{Prices $\rightarrow$ moments (direct)}
    \begin{itemize}
      \item Bakshi--Kapadia--Madan style moment integrals (2003) ($\sim$1858 cites)
      \item static replication using OTM option payoffs; numerical integration over strikes
    \end{itemize}
  \end{enumerate}

  \item \textbf{Citation-based ranking (high $\rightarrow$ low; by approximate cites)}
  \begin{enumerate}
    \item \textbf{Prices $\rightarrow$ pdf}

      \item analytic moments (family-specific; citations dispersed)
    \end{itemize}
  \end{enumerate}

  \item \textbf{Implemented methods (cross-sectional per date/area)}
  \begin{enumerate}
    \item \textbf{Most popular replication: Breeden--Litzenberger (two-sided)}
    \begin{itemize}
      \item file: \texttt{method\_main.py} $\rightarrow$ \texttt{moments\_main.csv}
      \item inputs: caps + floors + swaps
      \item construction: call-equivalent curve (puts converted via parity)
      \item smoothing: spline across strikes
      \item density proxy: second derivative; clipping negatives; renormalization
      \item moments: numerical integration on strike grid
      \item gates: monotone decreasing call curve; convexity in strike; parity residual filter
    \end{itemize}

    \item \textbf{Other replicated method: BKM direct moments (tail-stabilized)}
    \begin{itemize}
      \item file: \texttt{method\_moments\_direct\_bkm.py} $\rightarrow$ \texttt{moments\_direct\_bkm.csv}
      \item inputs: caps + floors + swaps
      \item preprocessing: deduplicate strikes; PV $\rightarrow$ undiscounted payoffs (divide by $B$)
      \item OTM selection: puts ($K\le K_\star$), calls ($K\ge K_\star$)
      \item gates: parity residual filter; monotonicity of OTM payoffs; min strikes отырып both wings
      \item stabilization: synthetic tail points (zero payoff); optional winsorization
      \item moments: raw moments (2--4) via strike integrals; central moments (var/skew/kurt)
    \end{itemize}

    \item \textbf{Parametric benchmark: Lognormal calibration}
    \begin{itemize}
      \item file: \texttt{method\_parametric\_lognormal.py} $\rightarrow$ \texttt{moments\_parametric\_lognormal.csv}
      \item assumption: $X=I_T/I_0$ lognormal
      \item calibration: fit $(\mu,\sigma)$ to OTM caps+floors; bounded $\sigma$; mean anchoring to $K_\star$
      \item outputs: analytic mean/variance/skew/kurt; RMSE diagnostic
      \item gates: parity residual filter; monotonicity of call/put PV curves; min strikes both wings
    \end{itemize}

    \item \textbf{Diagnostic benchmark: Normal fit to BL-proxy pdf}
    \begin{itemize}
      \item file: \texttt{method\_parametric\_normal.py} $\rightarrow$ \texttt{moments\_parametric\_normal.csv}
      \item steps: single-instrument price curve $\rightarrow$ spline $\rightarrow$ BL-proxy pdf $\rightarrow$ fit Normal on grid
      \item filters: convexity/monotonicity; boundary-mass threshold; variance sanity caps
      \item role: diagnostic only (positivity mismatch; truncation sensitivity)
    \end{itemize}
  \end{enumerate}

  \item \textbf{Comparison logic (identical inputs)}
  \begin{itemize}
    \item common keys: (date, area)
    \item intersection sample: retain rows where all compared methods return valid moments
    \item plots produced: time series of mean; time series of variance (log scale)
    \item diagnostics stored: strike span proxies; counts of OTM puts/calls; tail parameters; parity residual pass/fail
  \end{itemize}

  \item \textbf{Empirical interpretation: identification \& stability}
  \begin{enumerate}
    \item \textbf{Mean (first moment)}
    \begin{itemize}
      \item strong cross-method agreement: BL $\approx$ Lognormal $\approx$ BKM
      \item driver: near-ATM information; swap-anchored center ($K_\star$); parity enforcement
      \item Normal benchmark: flat/spiky behavior; diagnostic only
    \end{itemize}

    \item \textbf{Variance (second moment)}
    \begin{itemize}
      \item systematic gaps: BKM variance lower than BL/Lognormal in sparse-strike periods
      \item driver: tail dependence; missing far-OTM strikes; truncation sensitivity
      \item convergence episodes: improved strike coverage / liquidity (stress periods)
    \end{itemize}

    \item \textbf{Skewness (third central moment)}
    \begin{itemize}
      \item dependence: left-right tail balance (puts vs calls)
      \item instability sources: wing asymmetry; sparse tail quotes; regularization sensitivity
      \item interpretation: caution; conditional on balanced OTM coverage
    \end{itemize}

    \item \textbf{Kurtosis (fourth central moment)}
    \begin{itemize}
      \item dependence: extreme tails; highest sensitivity to missing strikes
      \item instability sources: truncation; tail-point assumptions; outlier quotes
      \item interpretation: very cautious; mainly qualitative tail-risk indicator
    \end{itemize}
  \end{enumerate}

  \item \textbf{Disagreement diagnosis}
  \begin{itemize}
    \item trigger: material divergence in mean/variance across methods
    \item checks: OTM strike counts; observed strike span; tail-point usage; parity residual filter; monotonicity failures
    \item primary cause: variance divergence aligned with limited far-OTM coverage; BKM downward bias
  \end{itemize}

  \item \textbf{Main conclusion: best way to work on moments}
  \begin{enumerate}
    \item \textbf{Mean}
    \begin{itemize}
      \item preferred: any method using both wings + parity + swap anchoring
      \item robustness: high; low tail dependence
    \end{itemize}
    \item \textbf{Variance}
    \begin{itemize}
      \item preferred: tail-regularized methods (parametric lognormal; smoothed two-sided BL)
      \item secondary: BKM with explicit tail stabilization; interpret as lower-bound under sparse tails
    \end{itemize}
    \item \textbf{Skewness}
    \begin{itemize}
      \item preferred: parametric/regularized methods; require balanced wing coverage
      \item caution: high sensitivity to asymmetric strike availability
    \end{itemize}
    \item \textbf{Kurtosis}
    \begin{itemize}
      \item preferred: parametric with strong tail structure; report with sensitivity flags
      \item caution: weakest identification; treat as qualitative
    \end{itemize}
  \end{enumerate}

\end{itemize}
